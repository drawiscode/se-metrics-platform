以下指令均在终端powershell里面运行

$BaseUrl = "http://127.0.0.1:8080"

# 1 创建仓库（不同步）
Invoke-RestMethod -Method Post -Uri "$BaseUrl/api/repos?full_name=octocat/Hello-World"

# 2 同步（写 snapshots/issues/pulls）
Invoke-RestMethod -Method Post -Uri "$BaseUrl/api/repos/1/sync"

# 3 查看落库 issues
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/issues?limit=20"
