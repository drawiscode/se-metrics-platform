//  microsoft/VibeVoice

$BaseUrl = "http://127.0.0.1:8080"

# 1 创建仓库（不同步）
Invoke-RestMethod -Method Post -Uri "$BaseUrl/api/repos?full_name=octocat/Hello-World"

# 删除仓库
Invoke-RestMethod -Method Delete -Uri "$BaseUrl/api/repos/1"

# 2 同步（写 snapshots/issues/pulls）
Invoke-RestMethod -Method Post -Uri "$BaseUrl/api/repos/1/sync"

# 3 查看落库 issues
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/issues?limit=20"


# 4 针对 items 数组逐条列出所有字段，方便查看对象属性（不被表格截断）
($resp = Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/issues").items | ForEach-Object { $_ | Format-List -Property * -Force }

Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/issues" | Select-Object -ExpandProperty items | ForEach-Object { $_ | ConvertTo-Json -Depth 10 -Compress } | Out-File -FilePath .\output\issues\issues_raw.ndjson -Encoding utf8

# ==============================
# CI 健康监控测试
# ==============================

# 1) 同步并拉取 CI 运行数据（可通过 ci_pages_count 控制页数）
Invoke-RestMethod -Method Post -Uri "$BaseUrl/api/repos/1/sync?ci_pages_count=2" | ConvertTo-Json -Depth 10


# 2) 查看 CI 运行明细（支持 status/conclusion 过滤）
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/ci/runs?limit=20" | ConvertTo-Json -Depth 10
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/ci/runs?status=completed&conclusion=failure&limit=20" | ConvertTo-Json -Depth 10


# 3) 查看 CI 健康摘要
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/ci/health" | ConvertTo-Json -Depth 10


# 3.1) 查看 CI 趋势（按天统计失败率）
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/ci/trend?days=14" | ConvertTo-Json -Depth 10


# 4) 查看风险告警（CI 异常会写入 alert_type=ci_pipeline_unhealthy）
Invoke-RestMethod -Method Get -Uri "$BaseUrl/api/repos/1/risk/alerts?limit=20" | ConvertTo-Json -Depth 10
